# Editor UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Grid34 editor UI: a Notion-style block editor with a page-tree sidebar, reactive read/write paths bound to `RepoStore`, a `DraftStore` write funnel into `CommitBuilder`/`Publisher`, and a `DbViewStore` query layer for a table-view "database" block — all wired through React Context with a per-type block component registry.

**Architecture:** Three reactive stores (`RepoStore` reused from storage, plus new `DraftStore` and `DbViewStore`) are provided to the component tree via React Context. A `blockComponentRegistry` maps `BlockType` to components (`ParagraphBlock`, `HeadingBlock`, `ListBlock`, `DatabaseBlock`). `PageTree` (sidebar) and `PageEditor` (main pane, with `LockedPageView` fallback) compose these. All local edits funnel through `DraftStore.stage` → debounced checkpoint → `CommitBuilder` → `Publisher`, mirroring the storage layer's write path. Built bottom-up and test-first: types → DraftStore → DbViewStore → block components → registry → PageTree/PageEditor/LockedPageView → Context providers → integration round-trip.

**Tech Stack:** React + Vite + TypeScript, Vitest, React Testing Library, `rxjs` (already a transitive dependency via `applesauce-core`; this plan adds it as a direct dependency since the editor stores construct `Observable`/`BehaviorSubject` directly), `sql.js`.

import type { EventTemplate, NostrEvent } from 'nostr-tools/pure'
import type { Page, PageTreeState } from '../../storage/repo/types'

/** Minimal read-side surface DraftStore needs from RepoStore: looking up a page's current state. */
export interface DraftRepoStore {
  getPage(pageId: string): Page | undefined
}

/** Matches CommitBuilder.buildPatchEventTemplate's shape from the storage layer. */
export interface CommitBuilder {
  buildPatchEventTemplate(options: { page: Page; repoId: string; cek: Uint8Array; createdAt: number }): EventTemplate
}

/** Matches Publisher.publishPatch's shape from the storage layer. */
export interface Publisher {
  publishPatch(
    template: EventTemplate,
    signer: unknown,
    relayPublisher: unknown,
    relayUrls: string[]
  ): Promise<NostrEvent>
}

export type DraftMap = Record<string, { pageId: string; content: Record<string, unknown> }>

export interface DraftStore {
  /** Stage a local edit to a block. Replaces any existing unflushed draft for
   * the same blockId, debouncing rapid keystrokes into one pending change. */
  stage(pageId: string, blockId: string, content: Record<string, unknown>): void

  /** Live observable of all currently-staged, not-yet-checkpointed drafts,
   * keyed by blockId. */
  drafts$: Observable<DraftMap>

  /** Force an immediate checkpoint of all staged drafts, bypassing the debounce. */
  flush(): Promise<void>
}

export interface CreateDraftStoreOptions {
  repoStore: DraftRepoStore
  commitBuilder: CommitBuilder
  publisher: Publisher
  signer: unknown
  relayPublisher: unknown
  relayUrls: string[]
  repoId?: string
  cek?: Uint8Array
  /** Debounce period (ms) of inactivity before an automatic checkpoint fires. */
  debounceMs?: number
}

export function createDraftStore(options: CreateDraftStoreOptions): DraftStore {
  const { repoStore } = options
  const draftsSubject = new BehaviorSubject<DraftMap>({})

  function stage(pageId: string, blockId: string, content: Record<string, unknown>): void {
    const current = draftsSubject.getValue()
    draftsSubject.next({
      ...current,
      [blockId]: { pageId, content },
    })
  }

  async function flush(): Promise<void> {
    // Checkpoint/publish logic added in Task 4.
    void repoStore
  }

  return {
    stage,
    drafts$: draftsSubject.asObservable(),
    flush,
  }
}

```typescript
export function createDraftStore(options: CreateDraftStoreOptions): DraftStore {
  const {
    repoStore,
    commitBuilder,
    publisher,
    signer,
    relayPublisher,
    relayUrls,
    repoId = 'workspace-repo',
    cek = new Uint8Array(32),
    debounceMs = 1000,
  } = options

  const draftsSubject = new BehaviorSubject<DraftMap>({})
  const timers = new Map<string, ReturnType<typeof setTimeout>>()

  function stage(pageId: string, blockId: string, content: Record<string, unknown>): void {
    const current = draftsSubject.getValue()
    draftsSubject.next({
      ...current,
      [blockId]: { pageId, content },
    })

    const existingTimer = timers.get(pageId)
    if (existingTimer) clearTimeout(existingTimer)
    timers.set(
      pageId,
      setTimeout(() => {
        timers.delete(pageId)
        void checkpointPage(pageId)
      }, debounceMs)
    )
  }

  /** Merges all staged drafts for `pageId` into its current Page, builds a
   * patch via CommitBuilder, and publishes it via Publisher. On success,
   * clears the checkpointed drafts from drafts$. */
  async function checkpointPage(pageId: string): Promise<void> {
    const drafts = draftsSubject.getValue()
    const draftEntriesForPage = Object.entries(drafts).filter(([, d]) => d.pageId === pageId)
    if (draftEntriesForPage.length === 0) return

    const page = repoStore.getPage(pageId)
    if (!page) return

    const updatedBlocks = page.blocks.map((block) => {
      const draft = drafts[block.id]
      if (draft && draft.pageId === pageId) {
        return { ...block, content: draft.content, updatedAt: Date.now() }
      }
      return block
    })
    const updatedPage: Page = { ...page, blocks: updatedBlocks, updatedAt: Date.now() }

    const template = commitBuilder.buildPatchEventTemplate({
      page: updatedPage,
      repoId,
      cek,
      createdAt: Math.floor(Date.now() / 1000),
    })

    await publisher.publishPatch(template, signer, relayPublisher, relayUrls)

    // Success: drop the checkpointed drafts for this page from drafts$.
    const remaining = { ...draftsSubject.getValue() }
    for (const [blockId] of draftEntriesForPage) {
      delete remaining[blockId]
    }
    draftsSubject.next(remaining)
  }

  async function flush(): Promise<void> {
    const pageIds = new Set(Object.values(draftsSubject.getValue()).map((d) => d.pageId))
    for (const pageId of pageIds) {
      const timer = timers.get(pageId)
      if (timer) {
        clearTimeout(timer)
        timers.delete(pageId)
      }
      await checkpointPage(pageId)
    }
  }

  return {
    stage,
    drafts$: draftsSubject.asObservable(),
    flush,
  }
}
```

    const drafts = draftsSubject.getValue()
    const draftEntriesForPage = Object.entries(drafts).filter(([, d]) => d.pageId === pageId)
    if (draftEntriesForPage.length === 0) return

    const page = repoStore.getPage(pageId)
    if (!page) return

      page: updatedPage,
      repoId,
      cek,
      createdAt: Math.floor(Date.now() / 1000),

    try {
      await publisher.publishPatch(template, signer, relayPublisher, relayUrls)

      if (timer) clearTimeout(timer)
      timers.set(
        pageId,
        setTimeout(() => {
          timers.delete(pageId)
          void checkpointPage(pageId)
        }, delay)
      )
      return
    }

    // Success: drop the checkpointed drafts for this page from drafts$ and
    // reset the retry counter.
    retryAttempts.delete(pageId)
    const remaining = { ...draftsSubject.getValue() }
    for (const [blockId] of draftEntriesForPage) {
      delete remaining[blockId]
    }
    draftsSubject.next(remaining)
  }

```typescript
import { BehaviorSubject, type Observable } from 'rxjs'
import { map } from 'rxjs/operators'
import type { Database } from 'sql.js'
import type { Row, ViewSpec } from '../types'

export interface DbViewStore {
  /** Reactive query: emits the current matching rows for a database block's
   * view. Re-emits whenever `notifyChanged(databaseId)` is called (the
   * Indexer signals this after applying new synced state — see DbViewStore
   * spec section). SQLite is read-only here: only SELECTs are issued. */
  observeRows(databaseId: string, view: ViewSpec): Observable<Row[]>

  /** Signals that the given database block's underlying rows changed in
   * SQLite, causing all active `observeRows` subscriptions for that
   * databaseId to re-query and re-emit. Called by the Indexer integration
   * (wired in the app shell, not by UI components directly). */
  notifyChanged(databaseId: string): void
}

/** Builds a parameterized SELECT against db_rows for the given ViewSpec. */
function buildQuery(view: ViewSpec): { sql: string; params: unknown[] } {
  const params: unknown[] = [view.databaseId]
  let sql = 'SELECT id, properties_json FROM db_rows WHERE database_block_id = ?'

  if (view.filter) {
    for (const [property, value] of Object.entries(view.filter)) {
      sql += ` AND json_extract(properties_json, '$.${property}') = ?`
      params.push(value)
    }
  }

  if (view.sort) {
    const direction = view.sort.direction === 'desc' ? 'DESC' : 'ASC'
    sql += ` ORDER BY json_extract(properties_json, '$.${view.sort.property}') ${direction}`
  }

  return { sql, params }
}

/** Projects raw db_rows columns into Row[], optionally narrowing to ViewSpec.columns. */
function runQuery(db: Database, view: ViewSpec): Row[] {
  const { sql, params } = buildQuery(view)
  const result = db.exec(sql, params as never)
  if (result.length === 0) return []

  return result[0].values.map((value) => {
    const id = value[0] as string
    const propertiesJson = value[1] as string
    const allValues = JSON.parse(propertiesJson) as Record<string, unknown>

    const values = view.columns
      ? Object.fromEntries(view.columns.map((col) => [col, allValues[col]]))
      : allValues

    return { id, values }
  })
}

export function createDbViewStore(db: Database): DbViewStore {
  // One BehaviorSubject<number> "tick" counter per databaseId — incrementing
  // it triggers all of that databaseId's observeRows subscriptions to re-run
  // their query, without each subscription managing its own change feed.
  const ticks = new Map<string, BehaviorSubject<number>>()

  function tickFor(databaseId: string): BehaviorSubject<number> {
    let subject = ticks.get(databaseId)
    if (!subject) {
      subject = new BehaviorSubject(0)
      ticks.set(databaseId, subject)
    }
    return subject
  }

  function observeRows(databaseId: string, view: ViewSpec): Observable<Row[]> {
    return tickFor(databaseId).pipe(map(() => runQuery(db, view)))
  }

  function notifyChanged(databaseId: string): void {
    const subject = ticks.get(databaseId)
    if (subject) subject.next(subject.getValue() + 1)
  }

  return { observeRows, notifyChanged }
}

import { useEffect, useState } from 'react'
import { useDbViewStore, useDraftStore } from '../contexts/storeContexts'
import type { BlockProps } from './ParagraphBlock'
import type { Row, ViewSpec } from '../types'

function readViewSpec(block: BlockProps['block']): ViewSpec {
  return block.content as unknown as ViewSpec
}

/** Database block rendered as a table view. Reads its own ViewSpec from
 * block.content, subscribes to DbViewStore.observeRows for live rows, and
 * routes cell edits back through DraftStore as block-content changes on the
 * owning database block — SQLite is a read-only derived cache, so row edits
 * never write to it directly. Pending edits accumulate in content.rowEdits
 * (rowId -> partial column updates) until the next checkpoint merges them. */
export function DatabaseBlock({ block, pageId }: BlockProps) {
  const draftStore = useDraftStore()
  const dbViewStore = useDbViewStore()
  const viewSpec = readViewSpec(block)
  const [rows, setRows] = useState<Row[]>([])

  useEffect(() => {
    const subscription = dbViewStore.observeRows(viewSpec.databaseId, viewSpec).subscribe(setRows)
    return () => subscription.unsubscribe()
  }, [dbViewStore, viewSpec.databaseId, JSON.stringify(viewSpec)])

  const columns = viewSpec.columns ?? (rows[0] ? Object.keys(rows[0].values) : [])

  function handleCellChange(rowId: string, column: string, value: string) {
    const existingRowEdits = (block.content.rowEdits as Record<string, Record<string, unknown>> | undefined) ?? {}
    const rowEdits = {
      ...existingRowEdits,
      [rowId]: { ...existingRowEdits[rowId], [column]: value },
    }
    draftStore.stage(pageId, block.id, { ...block.content, databaseId: viewSpec.databaseId, rowEdits })
  }

  return (
    <table>
      <thead>
        <tr>
          {columns.map((column) => (
            <th key={column} role="columnheader">{column}</th>
          ))}
        </tr>
      </thead>
      <tbody>
        {rows.map((row) => (
          <tr key={row.id}>
            {columns.map((column) => (
              <td key={column}>
                <input
                  type="text"
                  aria-label={`${column} for row ${row.id}`}
                  value={String(row.values[column] ?? '')}
                  onChange={(event) => handleCellChange(row.id, column, event.target.value)}
                />
              </td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  )
}

import { useEffect, useState } from 'react'
import { useRepoStore } from '../contexts/storeContexts'
import type { Page, PageTreeState } from '../../storage/repo/types'

export interface PageTreeProps {
  selectedPageId: string | null
  onSelectPage: (pageId: string) => void
}

function childrenOf(state: PageTreeState, parentId: string | null): Page[] {
  return Object.values(state.pages)
    .filter((page) => page.parentId === parentId)
    .sort((a, b) => a.order - b.order)
}

function PageNode({ page, state, selectedPageId, onSelectPage }: {
  page: Page
  state: PageTreeState
  selectedPageId: string | null
  onSelectPage: (pageId: string) => void
}) {
  const children = childrenOf(state, page.id)
  return (
    <li>
      <button
        type="button"
        aria-current={page.id === selectedPageId ? 'true' : undefined}
        onClick={() => onSelectPage(page.id)}
      >
        {page.title}
      </button>
      {children.length > 0 && (
        <ul>
          {children.map((child) => (
            <PageNode key={child.id} page={child} state={state} selectedPageId={selectedPageId} onSelectPage={onSelectPage} />
          ))}
        </ul>
      )}
    </li>
  )
}

/** Sidebar navigation. Subscribes to RepoStore's PageTreeState observable and
 * renders an expandable tree (parent/child via parentId, ordered via order),
 * emitting onSelectPage when a page is clicked. Purely presentational over
 * RepoStore data — does not read/write through DraftStore. */
export function PageTree({ selectedPageId, onSelectPage }: PageTreeProps) {
  const repoStore = useRepoStore()
  const [state, setState] = useState<PageTreeState>({ pages: {} })

  useEffect(() => {
    const subscription = repoStore.pageTree$.subscribe(setState)
    return () => subscription.unsubscribe()
  }, [repoStore])

  const roots = childrenOf(state, null)

  return (
    <nav aria-label="Page tree">
      <ul>
        {roots.map((page) => (
          <PageNode key={page.id} page={page} state={state} selectedPageId={selectedPageId} onSelectPage={onSelectPage} />
        ))}
      </ul>
    </nav>
  )
}

import { useEffect, useState } from 'react'
import { useRepoStore } from '../contexts/storeContexts'
import { blockComponentRegistry } from '../blocks/registry'
import { LockedPageView } from './LockedPageView'
import type { Page } from '../../storage/repo/types'

export interface PageEditorProps {
  pageId: string
}

interface PageObservation {
  status: 'loading' | 'ready' | 'locked'
  page?: Page
}

/** Main editing pane. Subscribes to RepoStore for the selected page; renders
 * a transient "decrypting…" state while loading, LockedPageView if the page
 * is undecryptable, or the block tree (via blockComponentRegistry) when
 * ready. Never calls CommitBuilder/Publisher directly — block components
 * stage edits through DraftStore themselves. */
export function PageEditor({ pageId }: PageEditorProps) {
  const repoStore = useRepoStore()
  const [observation, setObservation] = useState<PageObservation>({ status: 'loading' })

  useEffect(() => {
    setObservation({ status: 'loading' })
    const subscription = repoStore.observePage(pageId).subscribe(setObservation)
    return () => subscription.unsubscribe()
  }, [repoStore, pageId])

  if (observation.status === 'loading') {
    return <p role="status">Decrypting…</p>
  }

  if (observation.status === 'locked' || !observation.page) {
    return <LockedPageView pageId={pageId} pageTitle={observation.page?.title ?? ''} />
  }

  const page = observation.page

  return (
    <article aria-label={page.title}>
      <h1>{page.title}</h1>
      {page.blocks
        .slice()
        .sort((a, b) => a.order - b.order)
        .map((block) => {
          const Component = blockComponentRegistry[block.type as keyof typeof blockComponentRegistry]
          if (!Component) return null
          return <Component key={block.id} block={block} pageId={pageId} />
        })}
    </article>
  )
}
```
